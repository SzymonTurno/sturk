249static inline struct Vertegs*
250vx_inscirq(struct Vertegs* cirq, struct Vertegs* entry, int pos)
251{
252 const size_t next = VX_LIST_NEXT;
253 const size_t prev = VX_LIST_PREV;
254 struct Vertegs* v = NULL;
255
256 VX_ENSURE_MEM(entry);
257 if (!cirq) {
258 entry->nbor[next] = entry;
259 entry->nbor[prev] = entry;
260 return entry;
261 }
262
263 if (pos > 0)
264 v = vx_walk(cirq, next, pos);
265 else if (pos < -1)
266 v = vx_walk(cirq, prev, -(pos + 1));
267 else
268 v = cirq;
269 entry->nbor[next] = v;
270 entry->nbor[prev] = v->nbor[prev];
271 v->nbor[prev] = entry;
272 entry->nbor[prev]->nbor[next] = entry;
273 return pos ? cirq : entry;
274}

286static inline struct Vertegs* vx_remcirq(struct Vertegs** cirqp, int pos)
287{
288 const size_t next = VX_LIST_NEXT;
289 const size_t prev = VX_LIST_PREV;
290 struct Vertegs* ret = NULL;
291
292 VX_ENSURE_MEM(cirqp);
293 ret = *cirqp;
294 if (pos > 0)
295 ret = vx_walk(ret, next, pos);
296 else if (pos < 0)
297 ret = vx_walk(ret, prev, -pos);
298 ret->nbor[next]->nbor[prev] = ret->nbor[prev];
299 ret->nbor[prev]->nbor[next] = ret->nbor[next];
300 if (ret == ret->nbor[next])
301 *cirqp = NULL;
302 else if (ret == *cirqp)
303 *cirqp = ret->nbor[next];
304 return ret;
305}

351static inline struct Vertegs* vx_inorderfirst(struct Vertegs* v)
352{
353 struct Vertegs* p = NULL;
354
355 VX_ENSURE_MEM(v);
356 for (;;) {
357 if (v->nbor[VX_RB_LEFT]) {
358 v = v->nbor[VX_RB_LEFT];
359 } else {
360 p = vx_unmark(v->nbor[VX_RB_PARCOL]);
361 if (p && (p->nbor[VX_RB_RIGHT] == v))
362 v = p;
363 else
364 break;
365 }
366 }
367 return v;
368}

379static inline struct Vertegs* vx_inordernext(struct Vertegs* v)
380{
381 const size_t left = VX_RB_LEFT;
382 const size_t right = VX_RB_RIGHT;
383 struct Vertegs* p = NULL;
384
385 VX_ENSURE_MEM(v);
386 if (v->nbor[right])
387 p = vx_walk(v->nbor[right], left, -1);
388 else
389 while ((p = vx_unmark(v->nbor[VX_RB_PARCOL])) &&
390 v == p->nbor[right])
391 v = p;
392 return p;
393}

404static inline struct Vertegs* vx_preordernext(struct Vertegs* v)
405{
406 const size_t left = VX_RB_LEFT;
407 const size_t right = VX_RB_RIGHT;
408 struct Vertegs* p = NULL;
409
410 VX_ENSURE_MEM(v);
411 if (v->nbor[left])
412 return v->nbor[left];
413
414 if (v->nbor[right])
415 return v->nbor[right];
416
417 while ((p = vx_unmark(v->nbor[VX_RB_PARCOL]))) {
418 if (v != p->nbor[right] && p->nbor[right])
419 break;
420 v = p;
421 }
422 return p ? p->nbor[right] : NULL;
423}

434static inline struct Vertegs* vx_postorderfirst(struct Vertegs* v)
435{
436 VX_ENSURE_MEM(v);
437 v = vx_walksafe(v, VX_RB_PARCOL, -1);
438 for (;;) {
439 if (v->nbor[VX_RB_LEFT])
440 v = v->nbor[VX_RB_LEFT];
441 else if (v->nbor[VX_RB_RIGHT])
442 v = v->nbor[VX_RB_RIGHT];
443 else
444 break;
445 }
446 return v;

496static inline struct Vertegs*
497vx_rotleftrb(struct Vertegs* v, struct Vertegs* root)
498{
499 struct Vertegs* p = vx_unmark(v->nbor[VX_RB_PARCOL]);
500 struct Vertegs* y = v->nbor[VX_RB_RIGHT];
501 struct Vertegs** tmp = NULL;
502
503 VX_ENSURE_MEM(y);
504 tmp = &y->nbor[VX_RB_LEFT];
505 v->nbor[VX_RB_RIGHT] = *tmp;
506 if (*tmp) {
507 tmp = &(*tmp)->nbor[VX_RB_PARCOL];
508 *tmp = vx_mark(v, (intptr_t)*tmp);
509 }
510
511 tmp = &y->nbor[VX_RB_PARCOL];
512 *tmp = vx_mark(p, (intptr_t)*tmp);
513 if (!p)
514 root = y;
515 else if (v == p->nbor[VX_RB_LEFT])
516 p->nbor[VX_RB_LEFT] = y;
517 else
518 p->nbor[VX_RB_RIGHT] = y;
519
520 y->nbor[VX_RB_LEFT] = v;
521 tmp = &v->nbor[VX_RB_PARCOL];
522 *tmp = vx_mark(y, (intptr_t)*tmp);
523 return root;
524}

580static inline struct Vertegs*
581vx_insrebalrb(struct Vertegs* root, struct Vertegs* v)
582{
583 VX_ENSURE_MEM(v);
584 for (struct Vertegs *p = NULL, *g = NULL, *u = NULL;;) {
585 p = vx_unmark(v->nbor[VX_RB_PARCOL]);
586 if (vx_getrbcolor(p) == VX_RB_BLACK) {
587 /* Case 0: done. */
588 if (!p) {
589 v->nbor[VX_RB_PARCOL] = vx_mark(
590 v->nbor[VX_RB_PARCOL], VX_RB_BLACK);
591 root = v;
592 }
593 break;
594 }
595
596 /*
597 * P is red and, as only black nodes can be nil or root, thus:
598 * a) it cannot be nil,
599 * b) its parent (G) cannot be nil.
600 */
601 g = vx_unmark(p->nbor[VX_RB_PARCOL]);
602 u = (p == g->nbor[VX_RB_LEFT]) ? g->nbor[VX_RB_RIGHT]
603 : g->nbor[VX_RB_LEFT];
604 if (vx_getrbcolor(u) == VX_RB_RED) {
605 /* Case 1: recolor. */
606 p->nbor[VX_RB_PARCOL] =
607 vx_mark(p->nbor[VX_RB_PARCOL], VX_RB_BLACK);
608 u->nbor[VX_RB_PARCOL] =
609 vx_mark(u->nbor[VX_RB_PARCOL], VX_RB_BLACK);
610 g->nbor[VX_RB_PARCOL] =
611 vx_mark(g->nbor[VX_RB_PARCOL], VX_RB_RED);
612 v = g;
613 continue;
614 }
615
616 /*
617 * Case 2: rotate and recolor.
618 * Go to step 2.0 or 2.1?
619 */
620 if (p == g->nbor[VX_RB_LEFT]) {
621 /*
622 * Step 2.0: left.
623 * Is N inner grandchild of G?
624 */
625 if (v == p->nbor[VX_RB_RIGHT]) {
626 /*
627 * N is inner grandchild of G.
628 * Step 2.0.0: left-right.
629 */
630 root = vx_rotleftrb(p, root);
631 v = p;
632 p = vx_unmark(v->nbor[VX_RB_PARCOL]);
633 /* Go to step 2.0.1. */
634 }
635 /* Step 2.0.1: left-left. */
636 root = vx_rotrightrb(g, root);
637 /* Go to step 2.2. */
638 } else {
639 /*
640 * Step 2.1: right.
641 * Is N inner grandchild of G?
642 */
643 if (v == p->nbor[VX_RB_LEFT]) {
644 /*
645 * N is inner grandchild of G.
646 * Step 2.1.0: right-left.
647 */
648 root = vx_rotrightrb(p, root);
649 v = p;
650 p = vx_unmark(v->nbor[VX_RB_PARCOL]);
651 /* Go to step 2.1.1. */
652 }
653 /* Step 2.1.1: right-right. */
654 root = vx_rotleftrb(g, root);
655 /* Go to step 2.2. */
656 }
657
658 /* Step 2.2: swap colors. */
659 vx_swaprbcolors(p, g);
660 v = p;
661 }
662 VX_ASSERT(root);
663 return root;
664}
